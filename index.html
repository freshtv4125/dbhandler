Installation


// Include the DbHandler class
require_once 'DbHandler.php';

// Initialize the connection
$db = DbHandler::getInstance(
    'localhost',     // host
    'username',      // username
    'password',      // password
    'database_name'  // database name
);

Database Management

Creating a Database

createDatabase($name, $collation = 'utf8mb4_unicode_ci')

Creates a new database with the specified name and collation.


$db->createDatabase('my_application');

// With custom collation
$db->createDatabase('my_application', 'utf8mb4_general_ci');

Listing Databases

listDatabases()

Returns an array of all databases on the server.


$databases = $db->listDatabases();
foreach ($databases as $database) {
    echo $database . "\n";
}

Table Management

Creating Tables

createTable($tableName, array $columns, $engine = 'InnoDB')

Creates a new table with the specified columns and constraints.


$db->createTable('users', [
    'id' => [
        'type' => 'int',
        'length' => 11,
        'primary' => true,
        'autoIncrement' => true
    ],
    'username' => [
        'type' => 'varchar',
        'length' => 255,
        'unique' => true,
        'nullable' => false
    ],
    'email' => [
        'type' => 'varchar',
        'length' => 255,
        'unique' => true
    ],
    'created_at' => [
        'type' => 'timestamp',
        'default' => 'CURRENT_TIMESTAMP'
    ]
]);

Data Manipulation

Inserting Records

insert($table, array $data)

Inserts a single record into the specified table.


$db->insert('users', [
    'username' => 'john_doe',
    'email' => 'john@example.com',
    'created_at' => date('Y-m-d H:i:s')
]);

Batch Insert

insertBatch($table, array $rows)

Efficiently inserts multiple records in a single query.


$users = [
    ['username' => 'user1', 'email' => 'user1@example.com'],
    ['username' => 'user2', 'email' => 'user2@example.com'],
    ['username' => 'user3', 'email' => 'user3@example.com']
];
$db->insertBatch('users', $users);

Transactions

Transaction Management

Manage database transactions with support for nested transactions.


try {
    $db->beginTransaction();
    
    $db->insert('users', ['username' => 'john']);
    
    $db->beginTransaction(); // Nested transaction
    $db->insert('profiles', ['user_id' => 1]);
    $db->commit();
    
    $db->commit();
} catch (Exception $e) {
    $db->rollback();
    throw $e;
}
